import sqlite3, hashlib, base64, time, random, string from cryptography.fernet import Fernet from kivy.app import App from kivy.uix.screenmanager import ScreenManager, Screen, FadeTransition from kivy.uix.boxlayout import BoxLayout from kivy.uix.button import Button from kivy.uix.textinput import TextInput from kivy.uix.label import Label from kivy.uix.scrollview import ScrollView from kivy.uix.gridlayout import GridLayout from kivy.uix.spinner import Spinner from kivy.graphics import Color, RoundedRectangle from datetime import datetime # --- CONFIG STYLE --- COULEUR_FOND = (0.05, 0.05, 0.07, 1) # Noir Profond COULEUR_ACCENT = (0, 0.6, 1, 1) # Bleu Électrique COULEUR_BOUTON = (0.15, 0.15, 0.2, 1) # Gris Anthracite COULEUR_TEXTE = (0.9, 0.9, 0.9, 1) # --- LOGIQUE SÉCURITÉ (Inchangée) --- def generer_cle(password): hash_gen = hashlib.sha256(password.encode()).digest() return base64.urlsafe_b64encode(hash_gen) def initialiser_db(): conn = sqlite3.connect('mon_coffre_pro.db') cursor = conn.cursor() cursor.execute('''CREATE TABLE IF NOT EXISTS secrets (id INTEGER PRIMARY KEY AUTOINCREMENT, titre TEXT, contenu BLOB, categorie TEXT, date_modif TEXT, etat TEXT DEFAULT 'actif')''') conn.commit() return conn # --- BOUTON PERSONNALISÉ (Coins arrondis) --- class ModernButton(Button): def __init__(self, color=COULEUR_BOUTON, **kwargs): super().__init__(**kwargs) self.background_normal = "" self.background_color = (0,0,0,0) # On cache le fond par défaut self.color = COULEUR_TEXTE self.font_name = "Roboto" self.btn_color = color self.bind(pos=self.update_canvas, size=self.update_canvas) def update_canvas(self, *args): self.canvas.before.clear() with self.canvas.before: Color(*self.btn_color) RoundedRectangle(pos=self.pos, size=self.size, radius=[15,]) # --- ÉCRAN DE CONNEXION --- class LoginScreen(Screen): def __init__(self, **kwargs): super().__init__(**kwargs) self.tentatives = 0 self.fin_blocage = 0 with self.canvas.before: Color(*COULEUR_FOND) RoundedRectangle(pos=(0,0), size=(2000, 2000)) # Fond layout = BoxLayout(orientation='vertical', padding=80, spacing=30) self.status_label = Label(text="PROTECTION\nACTIVÉE", font_size=32, bold=True, halign="center", color=COULEUR_ACCENT) layout.add_widget(self.status_label) self.mdp_input = TextInput(hint_text="Code d'accès", password=True, multiline=False, size_hint_y=None, height=120, halign="center", background_color=(0.1, 0.1, 0.15, 1), foreground_color=(1,1,1,1), cursor_color=COULEUR_ACCENT) layout.add_widget(self.mdp_input) btn_ouvrir = ModernButton(text="DÉVERROUILLER", color=COULEUR_ACCENT, size_hint_y=None, height=130) btn_ouvrir.bind(on_press=self.verifier) layout.add_widget(btn_ouvrir) self.add_widget(layout) def verifier(self, instance): maintenant = time.time() if maintenant < self.fin_blocage: reste = int(self.fin_blocage - maintenant) self.status_label.text = f"SÉCURITÉ ACTIVE\n({reste}s)" return if self.mdp_input.text == "101225": self.tentatives = 0 app = App.get_running_app() app.cipher = Fernet(generer_cle(self.mdp_input.text)) self.manager.current = 'main' else: self.tentatives += 1 self.mdp_input.text = "" if self.tentatives >= 3: delai = (self.tentatives - 2) * 300 self.fin_blocage = maintenant + delai self.status_label.text = f"ALERTE INTRAUSION\n{delai//60}min" else: self.status_label.text = f"CODE ERRONE\n({3 - self.tentatives} essais)" # --- ÉCRAN PRINCIPAL --- class MainScreen(Screen): def on_enter(self): self.refresh() def refresh(self): self.clear_widgets() with self.canvas.before: Color(*COULEUR_FOND) RoundedRectangle(pos=(0,0), size=(2000, 2000)) layout = BoxLayout(orientation='vertical', padding=30, spacing=20) # Filtre stylé self.filtre = Spinner(text='Tous les secrets', values=('Toutes', 'Personnel', 'Travail', 'Jeux', 'Corbeille'), size_hint_y=None, height=110, background_color=COULEUR_BOUTON, color=COULEUR_TEXTE) self.filtre.bind(text=lambda x, y: self.afficher_liste(self.grid_secrets)) layout.add_widget(self.filtre) scroll = ScrollView() self.grid_secrets = GridLayout(cols=1, spacing=15, size_hint_y=None) self.grid_secrets.bind(minimum_height=self.grid_secrets.setter('height')) self.afficher_liste(self.grid_secrets) scroll.add_widget(self.grid_secrets) layout.add_widget(scroll) btn_add = ModernButton(text="+ AJOUTER UN SECRET", color=COULEUR_ACCENT, size_hint_y=None, height=130) btn_add.bind(on_press=lambda x: setattr(self.manager, 'current', 'add')) layout.add_widget(btn_add) self.add_widget(layout) def afficher_liste(self, grid): grid.clear_widgets() conn = initialiser_db() cursor = conn.cursor() cursor.execute('SELECT id, titre, contenu, categorie, date_modif, etat FROM secrets') app = App.get_running_app() for id_n, titre, cont_chiffre, cat, date, etat in cursor.fetchall(): if self.filtre.text == 'Corbeille' and etat != 'poubelle': continue if self.filtre.text != 'Corbeille' and etat == 'poubelle': continue if self.filtre.text not in ['Toutes', 'Corbeille', 'Tous les secrets'] and cat != self.filtre.text: continue try: txt = app.cipher.decrypt(cont_chiffre).decode() # On ne montre qu'un aperçu apercu = (txt[:20] + '...') if len(txt) > 20 else txt info = f"[b]{titre.upper()}[/b]\n[size=14]{cat} - {date}[/size]\n{apercu}" btn = ModernButton(text=info, size_hint_y=None, height=220, markup=True) btn.bind(on_release=lambda x, i=id_n, e=etat: self.changer_etat(i, e)) grid.add_widget(btn) except: pass def changer_etat(self, id_n, etat_actuel): nouvel_etat = 'poubelle' if etat_actuel == 'actif' else 'actif' conn = initialiser_db() conn.cursor().execute('UPDATE secrets SET etat = ? WHERE id = ?', (nouvel_etat, id_n)) conn.commit() self.refresh() # --- ÉCRAN D'AJOUT --- class AddScreen(Screen): def __init__(self, **kwargs): super().__init__(**kwargs) with self.canvas.before: Color(*COULEUR_FOND) RoundedRectangle(pos=(0,0), size=(2000, 2000)) layout = BoxLayout(orientation='vertical', padding=40, spacing=20) self.titre = TextInput(hint_text="Titre", size_hint_y=None, height=110, multiline=False, background_color=(0.1, 0.1, 0.15, 1), foreground_color=(1,1,1,1)) self.cat = Spinner(text='Personnel', values=('Personnel', 'Travail', 'Jeux'), size_hint_y=None, height=110, background_color=COULEUR_BOUTON) self.cont = TextInput(hint_text="Contenu secret", multiline=True, background_color=(0.1, 0.1, 0.15, 1), foreground_color=(1,1,1,1)) btn_gen = ModernButton(text="🎲 GÉNÉRER UN MOT DE PASSE", size_hint_y=None, height=110) btn_gen.bind(on_press=self.generer_mdp) btn_save = ModernButton(text="ENREGISTRER", color=COULEUR_ACCENT, size_hint_y=None, height=130) btn_save.bind(on_press=self.sauver) layout.add_widget(self.titre) layout.add_widget(self.cat) layout.add_widget(btn_gen) layout.add_widget(self.cont) layout.add_widget(btn_save) btn_back = Button(text="RETOUR", background_color=(0,0,0,0), color=COULEUR_ACCENT) btn_back.bind(on_press=lambda x: setattr(self.manager, 'current', 'main')) layout.add_widget(btn_back) self.add_widget(layout) def generer_mdp(self, instance): caract = string.ascii_letters + string.digits + "!@#$%" self.cont.text = "".join(random.choice(caract) for _ in range(14)) def sauver(self, instance): app = App.get_running_app() if self.titre.text and self.cont.text: cont_chiffre = app.cipher.encrypt(self.cont.text.encode()) date_now = datetime.now().strftime("%d/%m %H:%M") conn = initialiser_db() conn.cursor().execute('INSERT INTO secrets (titre, contenu, categorie, date_modif) VALUES (?, ?, ?, ?)', (self.titre.text, cont_chiffre, self.cat.text, date_now)) conn.commit() self.titre.text = ""; self.cont.text = "" self.manager.current = 'main' class CoffreApp(App): def build(self): self.cipher = None sm = ScreenManager(transition=FadeTransition()) sm.add_widget(LoginScreen(name='login')) sm.add_widget(MainScreen(name='main')) sm.add_widget(AddScreen(name='add')) return sm if __name__ == '__main__': CoffreApp().run()
